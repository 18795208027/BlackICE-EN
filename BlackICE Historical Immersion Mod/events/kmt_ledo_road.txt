
add_namespace = ledo_road


country_event = {
	id = ledo_road.1
	hidden = yes
	
	trigger = {
		always = no
		tag = USA
		has_global_flag = KMT_ledo_road_started_construction
		NOT = {
			has_global_flag = KMT_ledo_road_finished
			has_global_flag = KMT_USA_withdraw_support
		}
		CHI = {
			has_capitulated = no
			exists = yes
			NOT = {
				has_war_with = USA
			}
		}
		USA = {
			has_capitulated = no
			exists = yes
			NOT = {
				has_war_with = CHI
			}
		}
		749 = {
			controller = {
				NOT = {
					has_war_with = CHI
					has_war_with = USA
				}
			}
		}
	}
	
	mean_time_to_happen = { days = 0 }
	
	immediate = {
		modify_global_flag = {
			flag = KMT_ledo_road_progress
			value = 1
		}
		if = {
			limit = {
				has_global_flag = {
					flag = KMT_ledo_road_progress
					value > 36
				}
			}
			set_global_flag = KMT_ledo_road_finished
			if = {
				limit = {
					CHI = {
						has_defensive_war = yes
					}
				}
				CHI = {
					add_ideas = KMT_ledo_road_idea
					country_event = { id = kmt.516 }
				}
				USA = {
					country_event = { id = kmt.519 }
				}
			}
			766 = {
				add_building_construction = {
					type = infrastructure
					level = 2
					instant_build = yes
				}
			}
			747 = {
				add_building_construction = {
					type = infrastructure
					level = 2
					instant_build = yes
				}
			}
		}
	}
	
	option = {}
}
