#Countries gaining a production bonus if the country of a license is in the same faction
country_event = {
	id = licenses.1
	hidden = yes
	
	trigger = {
		is_ai = no
		is_in_faction = yes
	}
	
	mean_time_to_happen = {
		days = 2
	}
	
	immediate = {
		#set_country_flag = in_faction
		every_country = {
			ROOT = {
				remove_relation_modifier = {
					target = PREV
					modifier = in_faction
				}
				add_relation_modifier = {
					target = PREV
					modifier = in_faction
				}
			}
		}
	}
	
	option = {
	
	}
}

country_event = {
	id = licenses.2
	hidden = yes
	
	trigger = {
		is_ai = no 
		NOT = { has_country_flag = recent_license_government_relation }
	}
	
	mean_time_to_happen = {
		days = 2
	}
	
	immediate = {
		set_country_flag = {
			flag = recent_license_government_relation
			value = 1 
			days = 30 
		}
		every_country = {
			limit = {
				if = {
					limit = {
						ROOT = {
							NOT = {
								democratic_govt = yes 
							}
						}
					}
					NOT = { has_government = ROOT }
				}
				else_if = {
					limit = {
						ROOT = { democratic_govt = yes }
					}
					democratic_govt = yes 
				}		
			}
			ROOT = {
				remove_relation_modifier = {
					target = PREV
					modifier = different_ideology
				}
				add_relation_modifier = {
					target = PREV
					modifier = different_ideology
				}
			}
		}
	}
	
	option = {
	
	}
}

country_event = {
	id = licenses.4
	title = licenses.4.t
	desc = licenses.4.desc
	
	is_triggered_only = yes
	
	option = { #Yes
		ai_chance = { base = 100 }
		name = "Agree to their offer"
		add_opinion_modifier = {
			target = FROM
			modifier = license_cooperation
		}
		add_relation_modifier = {
			target = ROOT
			modifier = close_cooperation
		}
	}
	
	option = { #No
		ai_chance = { base = 0 }
		name = "Reject their offer"
		FROM = { country_event = { id = licenses.6 days = 1 } }
	}
}

country_event = {
	id = licenses.5
	hidden = yes
	
	is_triggered_only = yes
	
	immediate = {
		clr_country_flag = clear_economy_license_relations
		if = {
			limit = {
				num_of_factories < 50 
			}
			every_country = {
				ROOT = { add_relation_modifier = { target = PREV 
					modifier = large_economy_1 }
				}
			}
		}
		else_if = {
			limit = {
				num_of_factories < 75 
			}
			every_country = {
				ROOT = { add_relation_modifier = { target = PREV 
					modifier = large_economy_2 }
				}
			}
