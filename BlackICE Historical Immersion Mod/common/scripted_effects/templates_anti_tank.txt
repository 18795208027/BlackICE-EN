	}

			NOT = {

				has_country_flag = ROLE:TD/FOCUS:HEAVY 
			}
		}
		set_country_flag = ROLE:AT/FOCUS:LINE_ANTI_TANK
		if = { limit = { check_variable = { EAI_max_armor_seen < 30 } }
			set_country_flag = { flag = AT_LINE_1 value = 1 days = 1 }
			if = { limit = { log = yes } log = "[This.GetName] | [GetMonth] | AT_LINE_1" }
		}
